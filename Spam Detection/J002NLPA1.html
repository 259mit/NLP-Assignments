5. Preprocessing 2 - Modifying the corpus¶

In [15]:
final_corpus = []
for i in range(len(corpus)):
    word = word_tokenize(corpus[i].lower())
    word = [stemmer.stem(y) for y in word if y not in trashitems]
    j = " ".join(word)
    final_corpus.append(j)
